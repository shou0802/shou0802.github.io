運算式輸入計算練習

# -*- coding: utf-8 -*-
"""
Created on Thu Nov 12 13:09:12 2020

@author: shou
"""

x = int(input("請輸入第一個整數:"))
y = int(input("請輸入第二個整數:"))
z = input("請輸入運算方式(+、-、*、/、//、%、**):")

if z == '+':
   print("計算結果為({}+{}={})".format(x,y,(x+y)))
elif z == '-':
   print("計算結果為({}-{}={})".format(x,y,(x-y))) 
elif z == '*':
   print("計算結果為({}*{}={})".format(x,y,(x*y))) 
elif z == '/':
    if y == 0:
      print("除數不可為0")    
    elif y > 0:  
      print("計算結果為({}/{}={})".format(x,y,(x/y))) 
elif z == '//':
    if y == 0:
      print("除數不可為0")    
    elif y > 0: 
      print("計算結果為({}//{}={})".format(x,y,(x//y)))
elif z == '%':
    if y == 0:
      print("除數不可為0")    
    elif y > 0: 
      print("計算結果為({}%{}={})".format(x,y,(x%y))) 
elif z == '**':
   print("計算結果為({}**{}={})".format(x,y,(x**y))) 
else:
    print("運算方式不合規則")
